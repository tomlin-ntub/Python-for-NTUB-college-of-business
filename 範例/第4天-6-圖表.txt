(2)散佈圖
#===================================
# 匯入模組
#===================================
import pandas as pd
import matplotlib.pyplot as plt

#----------------------------------
# 百貨資料
# (1)管道: (1:旅館/餐廳/咖啡館, 2:零售)
# (2)地區: (1:里斯本, 2:波爾圖, 3:其他)
# (3)生鮮產品的年度支出
# (4)奶製品的年度支出
# (5)雜貨產品的年度支出
# (6)冷凍產品的年度支出
# (7)洗滌劑和紙製品的年度支出
# (8)熟食產品年度消費
#----------------------------------
df = pd.read_csv('wholesale.csv', names=['d1', 'd2', 'd3', 'd4', 'd5', 'd6', 'd7', 'd8'])

#----------------------------------
# 選取資料
#----------------------------------
data_1_1 = df['d5'][(df['d2']==1)]
data_2_1 = df['d4'][(df['d2']==1)]

data_1_2 = df['d5'][(df['d2']==2)]
data_2_2 = df['d4'][(df['d2']==2)]

#---------------------------
# 圖形設定/中文設定
#---------------------------
plt.rcParams['font.sans-serif']=['SimHei']
fig = plt.figure()
sub = fig.add_subplot(111)

#---------------------------
# 設定圖標題
#---------------------------
plt.title('百貨資料')

#---------------------------
# 設定X軸/Y軸
#---------------------------
plt.xlabel('雜貨產品')
plt.ylabel('奶製品')

#---------------------------
# 設定散佈圖
#---------------------------
sub.plot(data_1_1, data_2_1, color='r', marker='d', linestyle='None',alpha=0.2)
sub.plot(data_1_2, data_2_2, color='g', marker='d', linestyle='None',alpha=0.2)

#---------------------------
# 設定資料說明
#---------------------------
plt.legend(['里斯本', '波爾圖'], loc='upper left')

#---------------------------
# 設定格線
#---------------------------
plt.grid(True)

#---------------------------
# 儲存圖形
#---------------------------
plt.show()





(3)折線圖
#====================================
# 匯入外部模組
#====================================
import pandas as pd
import matplotlib.pyplot as plt

#------------------------------------
# 讀入資料CSV檔
# 日期(0), 名稱(1),
# 開盤價(2), 高點(3),
# 低點(4), 收盤價(5), 成交量(6)
#------------------------------------
df = pd.read_csv('stocks.csv', names=['d0', 'd1', 'd2', 'd3', 'd4', 'd5', 'd6'])

#-----------------------------
# 選取資料
#-----------------------------
df2 = df[(df['d0']>='2001/01/01')]

#-----------------------------
# 有條件的選出資料
#-----------------------------
data_1_A = pd.to_datetime(df2['d0'][df2['d1']=='A'], format='%Y-%m-%d')
data_2_A = df2['d2'][df2['d1']=='A']

data_1_B = pd.to_datetime(df2['d0'][df2['d1']=='B'], format='%Y-%m-%d')
data_2_B = df2['d2'][df2['d1']=='B']

#---------------------------
# 圖形設定/中文設定
#---------------------------
plt.rcParams['font.sans-serif']=['SimHei']
fig = plt.figure()
sub = fig.add_subplot(111)

#---------------------------
# 繪製資料
#---------------------------
sub.plot(data_1_A, data_2_A, 'b')
sub.plot(data_1_B, data_2_B, 'g')

#---------------------------
# 設定圖標題
#---------------------------
plt.title('2001股價')

#---------------------------
# 設定資料說明
#---------------------------
plt.legend(['A股', 'B股'], loc='upper right')
plt.xticks(rotation=20)

#---------------------------
# 設定格線
#---------------------------
plt.grid(True)

#---------------------------
# 儲存圖形
#---------------------------
plt.show()
